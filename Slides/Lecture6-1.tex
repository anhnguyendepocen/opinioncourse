\section{Cognitive Interviews}

\frame{
\frametitle{Cognitive Interviews}
\begin{itemize}\itemsep1em
\item Form groups of 4 people
\item Each person present their experiences
\item Discuss
\item Synthesize
\end{itemize}
}

\frame{
\frametitle{Synthesis}
\begin{itemize}\itemsep1em
\item How difficult is it to write opinion questions?
\item How well do people understand opinion questions?
\item What kind of opinion formating/reporting processes do people use?
\item How much should we trust surveys? And how much should we trust peoples' opinion self-reports?
\end{itemize}
}


\section{New Material}

\frame{
\frametitle{Context}
\begin{itemize}\itemsep1em
\item Dahl: People know their own interests
\item Downs: Rational decision making
\item Mansbridge: Anticipatory representation invites manipulation
\end{itemize}
}

\frame{
\frametitle{Where do opinions come from?}
\begin{itemize}\itemsep1em
\item Stable constraints
	\begin{itemize}
	\item Evolution
	\item Genetics/biology
	\item Personality
	\item Ideology and values
	\end{itemize}
\item Social environment
\end{itemize}
}

\frame{
\frametitle{Social influences on opinion}
\begin{itemize}\itemsep1em
\item Adolescent socialization
\item Social networks
\item Mass media
\end{itemize}
}

\frame{
\frametitle{Perspectives on media influence}
\begin{itemize}\itemsep1em
\item Hypodermic effects
\item Direct effects
\item Minimal effects
\item Something else?
\end{itemize}
}

\frame{
\frametitle{Methods of media influence}
\begin{itemize}\itemsep1em
\item Persuasion (or coercion)
\item Learning
\item Agenda-setting
\item Priming
\item Framing
\end{itemize}
}

\frame{
\frametitle{Discussion}
\begin{itemize}\itemsep1em
\item Does evidence of media influence mean citizens' are irrational, manipulable, or unaware of their interests?
\item By contrast, what are the benefits of media influence for citizens' decision making?
\end{itemize}
}

\frame{
\frametitle{Heuristics}
\begin{itemize}\itemsep1em
\item Heuristics are decision-making shortcuts
\item Downs: Heuristics help people be rational by minimizing effort
\item Lau and Redlawsk: Party, ideology, endorsements, viability, appearance
\item Deservingness heuristic means general opinions are primed by social experiences
\end{itemize}
}

\frame{
\frametitle{Discussion}
\begin{itemize}\itemsep1em
\item Under what conditions can we trust peoples' opinions?
\end{itemize}
}
