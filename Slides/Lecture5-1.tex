\frame{
\frametitle{Survey Sampling}
\begin{itemize}\itemsep1em
\item Assume opinion exists as a population parameter
\item Representative sampling identifies that parameter (with uncertainty)
\item Repeated sampling tends to produce similar results
\item Bias and variance
\end{itemize}
}


\frame{
\frametitle{Response and nonresponse}
\begin{itemize}\itemsep1em
\item Survey design systematically includes and excludes
	\begin{itemize}
	\item Exclusion based on sampling procedures
	\item Exclusion based on competence (Gallup)
	\end{itemize}
\item Survey participation decisions further produce exclusion
	\begin{itemize}
	\item Unit nonresponse
	\item Item nonresponse
	\end{itemize}
\end{itemize}
}

\frame<1>[label=questions]{
\frametitle{Asking questions}
\begin{itemize}\itemsep1em
\item Open ended versus close ended
\item<2-> Rating (ordered) versus ranking (categorical)
\item<3-> Unipolar versus bipolar
\item<4-> Number and order of response options
\item<5-> ``Don't know''/``No opinion'' options
\item<6-> ``Some people think\dots''
\end{itemize}
}

\frame{
\frametitle{Open ended question}
Do you favor or oppose the legalization of abortion?
}

\frame{
\frametitle{Close ended question}
Do you favor or oppose the legalization of abortion?
\begin{itemize}
\item Favor
\item Oppose
\end{itemize}
}

\againframe<2>{questions}

\frame[label=rating]{
\frametitle{Rating}
Which of the following best describes your opinion on abortion?
\begin{itemize}
\item It should be legal in all circumstances
\item It should be legal except in the third trimester of pregnancy
\item It should be legal only in the first trimester of pregnancy
\item It should be illegal except to save the life of the mother
\item It should be illegal in all circumstances
\end{itemize}
}

\frame{
\frametitle{Ranking}
Please rank the following from 1 for the option that best describes your opinion on abortion to 5 for the option that least well describes your opinion on abortion.
\begin{itemize}
\item It should be legal in all circumstances
\item It should be legal except in the third trimester
\item It should be legal except in the second and third trimesters
\item It should be illegal except to save the life of the mother
\item It should be illegal in all circumstances
\end{itemize}
}

\frame{
\frametitle{Question wording}
\begin{itemize}\itemsep1em
\item Stems that frame
\item Word choice
	\begin{itemize}
	\item Do you favor or oppose [abortion during the third trimester | so-called ``partial birth abortion'']?
	\item What is your personal opinion regarding whether or not [global warming | climate change] has been happening?
	\end{itemize}
\item Double barreled questions
	\begin{itemize}
	\item Do you favor legalized abortion because it gives women the right to choose?
	\item Do you oppose legalized abortion because it ends a human life?
	\end{itemize}
\end{itemize}
}
